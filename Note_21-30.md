static void dfs(int i,int currentDeep){
    visited[i] = true;
    currentDeep++;
    deep = Math.max(deep,currentDeep);
    for(j:adjList.get(i)){
        if(visited[j]==false){
            dfs(j,currentDeep)
        }
    }
}
